Quick Sort Algorithm

How It Works:

The Quick Sort algorithm is a divide-and-conquer algorithm that sorts an array of elements by selecting a pivot element, partitioning the array around the pivot, and recursively sorting the subarrays.

  • Choose a pivot element from the array.
  • Partition the array around the pivot element.
  • Recursively sort the subarrays before and after the pivot element.

C++ Implementation:

            
#include <iostream>
using namespace std;

int partition(int arr[], int low, int high) {
    int pivot = arr[high];
    int i = (low - 1);

    for (int j = low; j <= high - 1; j++) {
        if (arr[j] < pivot) {
            i++;
            swap(arr[i], arr[j]);
        }
    }
    swap(arr[i + 1], arr[high]);
    return (i + 1);
}

void quickSort(int arr[], int low, int high) {
    if (low < high) {
        int pivot = partition(arr, low, high);

        quickSort(arr, low, pivot - 1);
        quickSort(arr, pivot + 1, high);
    }
}

void printArray(int arr[], int size) {
    for (int i = 0; i < size; i++)
        cout << arr[i] << " ";
    cout << endl;
}

int main() {
    int arr[] = { 10, 7, 8, 9, 1, 5 };
    int n = sizeof(arr) / sizeof(arr[0]);

    quickSort(arr, 0, n - 1);
    cout << "Sorted array: \n";
    printArray(arr, n);

    return 0;
}
